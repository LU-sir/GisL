46  AbstractSymbolizer* pSymbolizer;
47  std::string featureName;
48  while (!sldStream.atEnd()) {
49  QXmlStreamReader::TokenType token = sldStream.readNext();
50 
51  switch (token) {
52  case QXmlStreamReader::NoToken: {
53  break;
54  }
55  case QXmlStreamReader::Invalid: {
56  // mError = MError::GisLError::ErrXml;
57  mErrorMessage = sldStream.errorString().toStdString();

60  case QXmlStreamReader::StartElement: {
61

64  }
