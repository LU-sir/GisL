40  /*
41  * function float** GetRasterBand(int z):
42  * This function reads a band from a geotiff at a
43  * specified vertical level (z value, 1 ...
44  * n bands). To this end, the Geotiff's GDAL
45  * data type is passed to a switch statement,
46  * and the template function GetArray2D (see below)
47  * is called with the appropriate C++ data type.
48  * The GetArray2D function uses the passed-in C++
49  * data type to properly read the band data from
50  * the Geotiff, cast the data to float**, and return
51  * it to this function. This function returns that
52  * float** pointer.
53  */
54  switch (GDALGetRasterDataType(pmRasterBand)) {
55  case 0:
56  mErr = LayerErr::DataErr;
57  return;
58  case 1:
59  // GDAL GDT_Byte (-128 to 127) - unsigned char
60  GetArray2D<unsigned char>(GDT_Byte, 1);
61  case 2:
62  // GDAL GDT_UInt16 - short
63  GetArray2D<unsigned short>(GDT_UInt16, 2);
64  case 3:
65  // GDT_Int16
66  GetArray2D<short>(GDT_Int16, 2);
67  case 4:
68  // GDT_UInt32
69  GetArray2D<unsigned int>(GDT_UInt32, 4);
70  case 5:
71  // GDT_Int32
72  GetArray2D<int>(GDT_Int32, 4);
73  case 6:
74  // GDT_Float32
75  GetArray2D<float>(GDT_Float32, 4);
76  case 7:

103 void gisl::RasterBand::contrastEnhancement(ContrastEnhancementMethod m) {
104  switch (m) {
105  case ContrastEnhancementMethod::Normal: {
106  imgData = fData.cast<int>();
107  break;
108  }
109  case ContrastEnhancementMethod::StretchToRealMinMax: {
110  imgData =
111  ((fData -
112  Eigen::MatrixXf::Constant(ySize, xSize, (float)this->minimumValue)) *
113  256.0f / (this->maximumValue - this->minimumValue))
114  .cast<int>();
115  break;
116  }
117  case ContrastEnhancementMethod::StretchToCumulative96RealMinMax: {
118  double minPercent = 0;
119  double maxPercent = 1;
120  unsigned long long p1 = 0;
121  unsigned long long p2 = this->histogramBuckets - 1;
122  while (minPercent < 0.02) {
123  minPercent += (double)this->histogramArray[p1++] /
124  double(this->xSize * this->ySize);
125  }
126  while (maxPercent > 0.98) {
127  maxPercent -= (double)this->histogramArray[p2--] /
128  double(this->xSize * this->ySize);
129  }
130  double gap = (this->maximumValue - this->minimumValue) /
131  (double)this->histogramBuckets;
132  imgData = ((fData - Eigen::MatrixXf::Constant(
133  ySize,
134  xSize,
135  float(minimumValue + (float)p1 * gap))) *
136  256.0 / float(minimumValue + (float)p2 * gap))
137  .cast<int>();
138  qDebug(
139  "2'per is: %f, 98'per is: %f",
140  float(minimumValue + (float)p1 * gap),
141  float(minimumValue + (float)p2 * gap));
142  imgData = (imgData.array() < 0)
143  .select(Eigen::MatrixXi::Constant(ySize, xSize, 0), imgData);
144  imgData =
145  (imgData.array() > 255)
146  .select(Eigen::MatrixXi::Constant(ySize, xSize, 255), imgData);
147  break;
148  }
149  case ContrastEnhancementMethod::StretchTo2StandardDeviation: {
150  imgData = ((fData - Eigen::MatrixXf::Constant(
151  ySize,
152  xSize,
153  float(meanValue - 2.0f * sigmaValue))) *
154  256.0f / sigmaValue)
155  .cast<int>();
156  imgData = (imgData.array() < 0)
157  .select(Eigen::MatrixXi::Constant(ySize, xSize, 0), imgData);
158  imgData =
159  (imgData.array() > 255)
160  .select(Eigen::MatrixXi::Constant(ySize, xSize, 255), imgData);
161  break;
162  }
163  default: {
164  imgData = fData.cast<int>();
165  break;
166  }
167  }

174 QPolygonF gisl::RasterBand::calHistogram() const {
175  QPolygonF polygonF{this->histogramBuckets};
176  double gap =
177  (this->maximumValue - this->minimumValue) / this->histogramBuckets;
178  double begin = this->minimumValue + 0.5 * gap;
179  for (int i = 0; i < this->histogramBuckets; ++i) {
180  polygonF << QPointF{begin + i * gap, (double)this->histogramArray[i]};
181  }
