18 void gisl::Trans2D::loadPosData(std::string_view sv) {
19  std::ifstream ifs;
20  ifs.open(sv.data(), std::ios::in);
21  if (ifs.fail())
22  return;
23  ifs.seekg(0, std::ifstream::end);
24  int fileSize = (int)ifs.tellg();
25  auto buffer = (char*)malloc(fileSize);
26  ifs.seekg(0, std::ifstream::beg);
27  ifs.read(buffer, fileSize);
28  int selCount = 0;
29  std::string val;
30  std::vector<float> inValues;
31  std::vector<float> refValues;
32  for (int i = 0; i < fileSize; ++i) {
33  if (buffer[i] == ',') {
34  if (selCount < 2) {
35  inValues.emplace_back(std::stof(val));
36  qDebug("inPos ( %d): %s", selCount, val.c_str());
37  } else {
38  refValues.emplace_back(std::stof(val));
39  qDebug("refPos ( %d): %s", selCount, val.c_str());
40  }
41  val = "";
42  selCount++;
43  continue;
44  }
45  if (buffer[i] == '\r') {
46  refValues.emplace_back(std::stof(val));
47  qDebug("refPos ( %d): %s", selCount, val.c_str());
48  i += 1;
49  selCount = 0;
50  continue;
51  }
52  val += buffer[i];
53  }
54  inPos = Eigen::Map<Eigen::MatrixXf>(
55  inValues.data(),
56  (int)inValues.size() / 2 - 1,
57  2);
58  refPos = Eigen::Map<Eigen::MatrixXf>(
59  refValues.data(),
60  (int)refValues.size() / 2 - 1,
61  2);
62  accuracyPos << inValues[inValues.size() - 2], inValues[inValues.size() - 1],
63  refValues[refValues.size() - 2], refValues[refValues.size() - 1];
64  free(buffer);
65 }

68  this->i2rAdjustA.col(0) = Eigen::MatrixXf::Constant(inPos.rows(), 1, 1);
69  this->i2rAdjustA.col(1) = refPos.col(0);

72  this->i2rAdjustA.col(4) = refPos.col(0).array() * refPos.col(1).array();

76  // refPos.col(1).array(), refPos.col(1).array().square();
77  this->i2rTrans = (i2rAdjustA.transpose() * i2rAdjustA).inverse() *

115 gisl::Trans2D::i2rTransPoint(const std::pair<int, int>& p) const noexcept {
116  auto q = std::make_pair((float)p.first, (float)p.second);
117  float xOut = this->i2rTrans(0, 0) + this->i2rTrans(1, 0) * q.first +
118  this->i2rTrans(2, 0) * q.second +
119  this->i2rTrans(3, 0) * q.first * q.first +
120  this->i2rTrans(4, 0) * q.first * q.second +
121  this->i2rTrans(5, 0) * q.second * q.second;
122  float yOut = this->i2rTrans(0, 1) + this->i2rTrans(1, 1) * q.first +
123  this->i2rTrans(2, 1) * q.second +
124  this->i2rTrans(3, 1) * q.first * q.first +
125  this->i2rTrans(4, 1) * q.first * q.second +
126  this->i2rTrans(5, 1) * q.second * q.second;
127  return std::pair<float, float>(
128  xOut - this->transRecMN.second,
129  yOut - this->transRecMN.first);
