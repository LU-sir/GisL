76 void DaDecoder::quickSort(int left, int right) {
77  if (left < right) {
78  int i = left, j = right;
79  DaUnit* pTempUnit = pUnits[left];
80  while (i < j) {

82  j--;
83  if (i < j)

85  while (i < j && DaUnit::isSmaller(*pUnits[i], *pTempUnit))
